9const USER_STACK_SIZE: usize = 4096;
10const KERNEL_STACK_SIZE: usize = 4096 * 2;
11const MAX_APP_NUM: usize = 16;
12const APP_BASE_ADDRESS: usize = 0x80400000;
13const APP_SIZE_LIMIT: usize = 0x20000;
14
15#[repr(align(4096))]
16struct KernelStack {
17    data: [u8; KERNEL_STACK_SIZE],
18}
19
20#[repr(align(4096))]
21struct UserStack {
22    data: [u8; USER_STACK_SIZE],
23}
24
25static KERNEL_STACK: KernelStack = KernelStack { data: [0; KERNEL_STACK_SIZE] };
26static USER_STACK: UserStack = UserStack { data: [0; USER_STACK_SIZE] };
27
28impl KernelStack {
29    fn get_sp(&self) -> usize {
30        self.data.as_ptr() as usize + KERNEL_STACK_SIZE
31    }
32    pub fn push_context(&self, cx: TrapContext) -> &'static mut TrapContext {
33        let cx_ptr = (self.get_sp() - core::mem::size_of::<TrapContext>()) as *mut TrapContext;
34        unsafe {
35            *cx_ptr = cx;
36        }
37        unsafe { cx_ptr.as_mut().unwrap() }
38    }
39}
40
41impl UserStack {
42    fn get_sp(&self) -> usize {
43        self.data.as_ptr() as usize + USER_STACK_SIZE
44    }
45}
46
47struct AppManager {
48    num_app: usize,
49    current_app: usize,
50    app_start: [usize; MAX_APP_NUM + 1],
51}
52
53impl AppManager {
54    pub fn current_app_size(&self) -> usize {
55        // when load one, current_app++ so use current_app..current_app -1
56        self.app_start[self.current_app] - self.app_start[self.current_app - 1]
57    }
58    pub fn print_app_info(&self) {
59        println!("[kernel] num_app = {}", self.num_app);
60        for i in 0..self.num_app {
61            println!(
62                "[kernel] app_{} [{:#x}, {:#x})",
63                i,
64                self.app_start[i],
65                self.app_start[i + 1]
66            );
67        }
68    }
69
70    fn load_app(&self, app_id: usize) {
71        if app_id >= self.num_app {
72            println!("All applications completed!");
73            shutdown(false);
74        }
75        println!("[kernel] Loading app_{}", app_id);
76        unsafe {
77            // clear app area
78            core::slice::from_raw_parts_mut(APP_BASE_ADDRESS as *mut u8, APP_SIZE_LIMIT).fill(0);
79            let app_src = core::slice::from_raw_parts(
80                self.app_start[app_id] as *const u8,
81                self.app_start[app_id + 1] - self.app_start[app_id],
82            );
83            let app_dst =
84                core::slice::from_raw_parts_mut(APP_BASE_ADDRESS as *mut u8, app_src.len());
85            app_dst.copy_from_slice(app_src);
86            // Memory fence about fetching the instruction memory
87            // It is guaranteed that a subsequent instruction fetch must
88            // observes all previous writes to the instruction memory.
89            // Therefore, fence.i must be executed after we have loaded
90            // the code of the next app into the instruction memory.
91            // See also: riscv non-priv spec chapter 3, 'Zifencei' extension.
92            asm!("fence.i");
93        }
94    }
95
96    pub fn get_current_app(&self) -> usize {
97        self.current_app
98    }
99
100    pub fn move_to_next_app(&mut self) {
101        self.current_app += 1;
102    }
103}
104
105lazy_static! {
106    static ref APP_MANAGER: UPSafeCell<AppManager> = unsafe {
107        UPSafeCell::new({
108            unsafe extern "C" {
109                safe fn _num_app();
110            }
111            let num_app_ptr = _num_app as usize as *const usize;
112            let num_app = num_app_ptr.read_volatile();
113            let mut app_start: [usize; MAX_APP_NUM + 1] = [0; MAX_APP_NUM + 1];
114            let app_start_raw: &[usize] =
115                core::slice::from_raw_parts(num_app_ptr.add(1), num_app + 1);
116            app_start[..=num_app].copy_from_slice(app_start_raw);
117            AppManager { num_app, current_app: 0, app_start }
118        })
119    };
120}

122/// init batch subsystem
123pub fn init() {
124    print_app_info();
125}
126/// get app range
127pub fn get_cur_app_range() -> Range<usize> {
128    APP_BASE_ADDRESS
129        ..APP_BASE_ADDRESS
130            + APP_MANAGER
131                .exclusive_access()
132                .current_app_size()
133}
134
135/// get user stack range
136pub fn get_user_stack_range() -> Range<usize> {
137    USER_STACK.get_sp() - USER_STACK_SIZE..USER_STACK.get_sp()
138}
139
140/// print apps info
141pub fn print_app_info() {
142    APP_MANAGER
143        .exclusive_access()
144        .print_app_info();
145}
146
147/// run next app
148pub fn run_next_app() -> ! {
149    let mut app_manager = APP_MANAGER.exclusive_access();
150    let current_app = app_manager.get_current_app();
151    app_manager.load_app(current_app);
152    app_manager.move_to_next_app();
153    drop(app_manager);
154    // before this we have to drop local variables related to resources manually
155    // and release the resources
156    unsafe extern "C" {
157        unsafe fn __restore(cx_addr: usize);
158    }
159    unsafe {
160        __restore(KERNEL_STACK.push_context(TrapContext::app_init_context(
161            APP_BASE_ADDRESS,
162            USER_STACK.get_sp(),
163        )) as *const _ as usize);
164    }
165    panic!("Unreachable in batch::run_current_app!");
166}
